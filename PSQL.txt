parte 1

CREATE TABLESPACE mid_term
DATAFILE 'mid_term.dbf' SIZE 40M
EXTENT MANAGEMENT LOCAL AUTOALLOCATE;

CREATE USER cchanci IDENTIFIED BY cchanci
DEFAULT TABLESPACE  mid_term
QUOTA UNLIMITED ON mid_term;

CREATE ROLE MINI_DBA;
GRANT CREATE TABLE, CREATE SEQUENCE TO MINI_DBA;

CREATE ROLE STUDENT;
GRANT CREATE SESSION, MINI_DBA TO STUDENT;

GRANT STUDENT TO CCHANCI;

parte 2
CREATE SEQUENCE DELITOS_SEQ
START With 100
MAXVALUE 200
INCREMENT BY 1
NOCYCLE;

CREATE SEQUENCE CAPTURAS_SEQ
START With 1000
MAXVALUE 2000
INCREMENT BY 1  
NOCYCLE;

CREATE TABLE DELITOS(
DELITO_ID INT PRIMARY KEY,
DELITO_NAME VARCHAR2(255) NOT NULL
);

CREATE TABLE CAPTURAS(
CAPTURA_ID INT PRIMARY KEY,
DIA DATE NOT NULL,
LUGAR_ID INT NOT NULL,
SOSPECHOSO_ID INT NOT NULL,
EDAD INT NOT NULL,
SEXO VARCHAR2(255)NOT NULL,
ESTADO_CIVIL VARCHAR2(255) NOT NULL,
CLASE_EMPLEADO VARCHAR2(255) NOT NULL,
PROFESION VARCHAR2(255) DEFAULT ('-0'),
ESCOLARIDAD VARCHAR2(255) DEFAULT ('-0'),
DEPARTAMENTO VARCHAR2(255)NOT NULL,
MUNICIPIO VARCHAR2(255)NOT NULL,
BARRIO VARCHAR2(255)NOT NULL,
ZONA VARCHAR2(255)NOT NULL,
CLASE_SITIO VARCHAR2(255)NOT NULL,
DELITO_ID INT NOT NULL,
CONSTRAINT fk_capturas_delito FOREIGN KEY(DELITO_ID) REFERENCES DELITOS(DELITO_ID)
);
